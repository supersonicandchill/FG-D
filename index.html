// ==================== AUDIO + D&amp;D FLAIR ====================

let audioContext;

let soundEnabled = true;


function initAudio() {

  if (!audioContext) audioContext = new (window.AudioContext || window.webkitAudioContext)();

}


function playSiren() {

  if (!soundEnabled || !audioContext) return;

  initAudio();

  const osc = audioContext.createOscillator();

  osc.type = 'sawtooth';

  osc.frequency.setValueAtTime(280, audioContext.currentTime);

  const gain = audioContext.createGain();

  gain.gain.value = 0.45;

  osc.connect(gain).connect(audioContext.destination);

  osc.start();

  let dir = 1;

  const wail = setInterval(() => {

    const f = osc.frequency.value;

    osc.frequency.setValueAtTime(Math.max(220, Math.min(650, f + dir * 160)), audioContext.currentTime);

    if (f > 600 || f < 240) dir *= -1;

  }, 90);

  setTimeout(() => { clearInterval(wail); osc.stop(); }, 1300);

}


function playBrakeScreech() {

  if (!soundEnabled || !audioContext) return;

  initAudio();

  const bufferSize = audioContext.sampleRate * 1.6;

  const buffer = audioContext.createBuffer(1, bufferSize, audioContext.sampleRate);

  const output = buffer.getChannelData(0);

  for (let i = 0; i < bufferSize; i++) output[i] = (Math.random() * 2 - 1) * (1 - i / bufferSize);

  const noise = audioContext.createBufferSource();

  noise.buffer = buffer;

  const filter = audioContext.createBiquadFilter();

  filter.type = 'lowpass';

  filter.frequency.setValueAtTime(1400, audioContext.currentTime);

  const gain = audioContext.createGain();

  gain.gain.setValueAtTime(0.7, audioContext.currentTime);

  gain.gain.exponentialRampToValueAtTime(0.001, audioContext.currentTime + 1.6);

  noise.connect(filter).connect(gain).connect(audioContext.destination);

  noise.start();

}


function playDiceRoll() {

  if (!soundEnabled || !audioContext) return;

  initAudio();

  // Classic D20 clatter

  const osc = audioContext.createOscillator();

  osc.frequency.setValueAtTime(120 + Math.random() * 300, audioContext.currentTime);

  const gain = audioContext.createGain();

  gain.gain.setValueAtTime(0.8, audioContext.currentTime);

  gain.gain.exponentialRampToValueAtTime(0.001, audioContext.currentTime + 0.8);

  osc.connect(gain).connect(audioContext.destination);

  osc.start();

}

// Drop handler with D&amp;D bounce

const container = document.getElementById('canvas-container');

container.addEventListener('dragover', e => e.preventDefault());

container.addEventListener('drop', e => {

  e.preventDefault();

  const type = e.dataTransfer.getData('text/plain');

  const rect = canvas.getElement().getBoundingClientRect();

  let x = e.clientX - rect.left;

  let y = e.clientY - rect.top;


  let obj;

  const baseProps = { left: x, top: y, selectable: true, hasControls: true };


  if (type.includes('victim')) {

    const colors = { low: '#22c55e', mod: '#eab308', high: '#f97316', confirmed: '#ef4444' };

    const key = type.split('-')[1] || 'low';

    obj = new fabric.Circle(Object.assign(baseProps, {

      radius: 24, fill: colors[key], stroke: '#ffdd00', strokeWidth: 5, label: type.toUpperCase()

    }));

  } else if (type === 'fire') {

    obj = new fabric.Text('🐉', Object.assign(baseProps, { fontSize: 58, label: 'CHAOS DRAGON' }));

  } else {

    const emoji = iconsData.find(i => i.type === type)?.emoji || '🔥';

    obj = new fabric.Text(emoji, Object.assign(baseProps, { fontSize: 52, label: type.toUpperCase() }));

  }


  obj.scale(0.2);

  canvas.add(obj);

  playBrakeScreech();


  obj.animate({ scaleX: 1, scaleY: 1 }, {

    duration: 320,

    onChange: canvas.renderAll.bind(canvas),

    easing: fabric.util.ease.easeOutElastic

  });

  canvas.setActiveObject(obj);

});


// Tool drawing (still works the same – now “spells”)

canvas.on('mouse:down', function (opt) {

  if (!currentTool) return;

  const pointer = canvas.getPointer(opt.e);

  let target = null;

  canvas.getObjects().forEach(obj => {

    if (!obj.label) return;

    const isEngine = currentTool === 'hose' && obj.label.includes('ENGINE');

    const isTruck = currentTool === 'ladder' && obj.label.includes('TRUCK');

    if (isEngine || isTruck) {

      const dist = Math.hypot(obj.left - pointer.x, obj.top - pointer.y);

      if (dist < 90) target = obj;

    }

  });


  if (target) {

    const offsetX = currentTool === 'hose' ? 45 : 35;

    const offsetY = currentTool === 'hose' ? 25 : 45;

    const startX = target.left + offsetX;

    const startY = target.top + offsetY;


    drawingLine = new fabric.Line([startX, startY, pointer.x, pointer.y], {

      stroke: currentTool === 'hose' ? '#ff5500' : '#ffdd00',

      strokeWidth: currentTool === 'hose' ? 16 : 10,

      strokeDashArray: currentTool === 'ladder' ? [10, 5] : null,

      strokeLineCap: 'round',

      label: currentTool.toUpperCase() + ' SPELL',

      selectable: true

    });

    canvas.add(drawingLine);

  }

});


canvas.on('mouse:move', function (opt) {

  if (drawingLine) {

    const pointer = canvas.getPointer(opt.e);

    drawingLine.set({ x2: pointer.x, y2: pointer.y });

    canvas.renderAll();

  }

});


canvas.on('mouse:up', function () {

  if (drawingLine) {

    drawingLine.setCoords();

    drawingLine = null;

    currentTool = null;

    canvas.defaultCursor = 'default';

  }

});


function activateTool(tool) {

  currentTool = tool;

  canvas.defaultCursor = 'crosshair';

}

}


// ==================== D&amp;D SIMULATION + DICE ====================

function rollD20(modifier = 0) {

  playDiceRoll();

  return Math.floor(Math.random() * 20) + 1 + modifier;

}


function runSimulation() {

  const objects = canvas.getObjects();

  let hasCommand = false;

  let victimCount = 0;

  let dragonSlain = false;


  objects.forEach(obj => {

    const lbl = (obj.label || '').toUpperCase();

    if (lbl.includes('COMMAND') || lbl.includes('PALADIN')) hasCommand = true;

    if (lbl.includes('VICTIM')) victimCount++;

    if (lbl.includes('ENGINE') || lbl.includes('DRAGON')) dragonSlain = true;

  });


  const initiativeRoll = rollD20(hasCommand ? 4 : 0); // +4 if Command Post placed (Kastros Tactical Gap bonus)


  let feedbackHTML = `

    <div class="narrative-header"><i class="fas fa-dice-d20"></i> INITIATIVE ROLL: <strong>${initiativeRoll}</strong></div>

    <p><strong>The Chaos Dragon stirs...</strong></p>

  `;


  if (initiativeRoll >= 15) {

    feedbackHTML += `<p style="color:#22c55e">🌟 Critical Success! Your FPODP size-up was legendary (Kastros Ch. 5). The party advances!</p>`;

  } else if (initiativeRoll >= 10) {

    feedbackHTML += `<p>✅ Solid roll. Tactical Gap bridged — crews are in position.</p>`;

  } else {

    feedbackHTML += `<p style="color:#ffcc00">⚠️ The Dragon breathes fire! Re-roll with better Command Post placement.</p>`;

  }


  if (victimCount > 0) feedbackHTML += `<p>👤 ${victimCount} civilians rescued — SIGNAL method applied like a true Cleric!</p>`;

  if (dragonSlain) feedbackHTML += `<p>🐉 Chaos Dragon defeated! Aggressive life-safety tactics unlocked (Book Ch. 6).</p>`;


  feedbackHTML += `<p style="margin-top:12px; font-size:0.85rem; opacity:0.8">XP Gained: ${Math.floor(Math.random()*30 + 20)} • Next level: Master FPODP</p>`;


  const fb = document.getElementById('feedback');

  fb.innerHTML = feedbackHTML;

  fb.style.display = 'block';

  setTimeout(() => { fb.style.display = 'none'; }, 8500);

}


function rollForChaos() {

  const roll = rollD20();

  const narrative = document.getElementById('narrative');

  

  let text = `<div class="narrative-header"><i class="fas fa-dice-d20"></i> CHAOS ROLL: <strong>${roll}</strong></div>`;

  

  if (roll >= 18) text += `<p>🌟 The gods smile! Perfect wind shift — ventilation spell succeeds automatically.</p>`;

  else if (roll >= 13) text += `<p>✅ Random event: Mayday call — but your RIC team is already in position. Good call!</p>`;

  else if (roll >= 8) text += `<p>⚔️ Minor complication: Fire doubles in size. Apply RECEO-VS immediately!</p>`;

  else text += `<p>💥 Critical Chaos! Flashover imminent. Pull back and re-size-up per Kastros Ten Commandments.</p>`;


  narrative.innerHTML = text;

  narrative.style.display = 'block';

  setTimeout(() => { narrative.style.display = 'none'; }, 7000);

}
